<script>
document.addEventListener('DOMContentLoaded', function () {

  }

  const submitBtn = document.getElementById('kpSubmitBtn');
  const policy = document.getElementById('kpPolicyCheck');
  const errorBox = document.getElementById('kpFormError');
  const successBox = document.getElementById('kpFormSuccess');

  function showError(text) {
    if (!errorBox) return;
    errorBox.textContent = text;
    errorBox.classList.add('is-visible');
  }

  function hideMessages() {
    if (errorBox) errorBox.classList.remove('is-visible');
    if (successBox) successBox.classList.remove('is-visible');
  }

  function getVisibleField(selectors) {
    for (const selector of selectors) {
      const el = customForm.querySelector(selector);
      if (el) return el;
    }
    return null;
  }

  function getTildaField(selectors) {
    for (const selector of selectors) {
      const el = tildaForm.querySelector(selector);
      if (el) return el;
    }
    return null;
  }

  function setNativeValue(element, value) {
    if (!element) return;

    const tag = element.tagName.toLowerCase();

    if (tag === 'input' || tag === 'textarea' || tag === 'select') {
      const prototype = Object.getPrototypeOf(element);
      const descriptor = Object.getOwnPropertyDescriptor(prototype, 'value');

      if (descriptor && descriptor.set) {
        descriptor.set.call(element, value);
      } else {
        element.value = value;
      }

      element.dispatchEvent(new Event('input', { bubbles: true }));
      element.dispatchEvent(new Event('change', { bubbles: true }));
      element.dispatchEvent(new Event('blur', { bubbles: true }));
      return;
    }

    element.value = value;
    element.dispatchEvent(new Event('input', { bubbles: true }));
    element.dispatchEvent(new Event('change', { bubbles: true }));
    element.dispatchEvent(new Event('blur', { bubbles: true }));
  }

  function debugTildaFields() {
    const all = tildaForm.querySelectorAll('input, textarea, select');
    const dump = [];
    all.forEach((el, i) => {
      dump.push({
        index: i,
        tag: el.tagName,
        name: el.getAttribute('name'),
        type: el.getAttribute('type'),
        value: el.value
      });
    });
    console.log('📋 Поля формы Тильды:', dump);
  }

  customForm.addEventListener('submit', function (e) {
    e.preventDefault();
    hideMessages();

    const visibleName = getVisibleField([

    ]);

    const area = visibleArea ? visibleArea.value.trim() : '';
    const comment = visibleComment ? visibleComment.value.trim() : '';

    console.log('Данные видимой формы:', { name, phone, area, comment });

    if (!name || !phone || !area) {
      showError('Заполните обязательные поля');
      return;
    }

    if (!policy || !policy.checked) {
      showError('Подтвердите согласие с политикой конфиденциальности');
      return;
    }

    debugTildaFields();

    const tildaName = getTildaField([
      'input[name="name"]',
      'input[name="Name"]',
      'input[type="text"]'
    ]);

    const tildaPhone = getTildaField([
      'input[name="phone"]',
      'input[name="Phone"]',
      'input[type="tel"]'
    ]);

    const tildaArea = getTildaField([
      'select[name="area"]',
      'select[name="Area"]',
      'input[name="area"]',
      'input[name="Area"]'
    ]);

    const tildaComment = getTildaField([
      'textarea[name="comment"]',
      'textarea[name="Comment"]',
      'input[name="comment"]',
      'input[name="Comment"]'
    ]);

    console.log('Найденные поля Тильды:', {
      tildaName,
      tildaPhone,
      tildaArea,
      tildaComment
    });

    if (!tildaName || !tildaPhone || !tildaArea || !tildaComment) {
      showError('Не найдены нужные поля в форме Тильды. Открой консоль F12.');
      return;
    }

    setNativeValue(tildaName, name);
    setNativeValue(tildaPhone, phone);
    setNativeValue(tildaArea, area);
    setNativeValue(tildaComment, comment);

      tildaName: tildaName.value,
      tildaPhone: tildaPhone.value,
      tildaArea: tildaArea.value,
      tildaComment: tildaComment.value
    });

    if (submitBtn) {
      submitBtn.disabled = true;
      submitBtn.classList.add('is-loading');
    }

      tildaForm.querySelector('button[type="submit"]') ||
      tildaForm.querySelector('input[type="submit"]') ||
      tildaForm.querySelector('.t-submit');

        submitBtn.disabled = false;
        submitBtn.classList.remove('is-loading');

        submitBtn.disabled = false;
        submitBtn.classList.remove('is-loading');
      }
      if (successBox) successBox.classList.add('is-visible');
